```python
def fibonacci(n):
    assert isinstance(n, int), "n must be an integer"
    assert n >= 0, "n must be non-negative"
    assert n <= 30, "n must be no greater than 30"

    if n == 0:
        return 0
    if n == 1:
        return 1

    a, b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b

    result = b
    assert result >= 0, "result must be non-negative"
    assert result == (fibonacci(n - 1) + fibonacci(n - 2)), "result must satisfy recurrence relation"
    return result


result = fibonacci(10)
assert result == 55, f"Expected 55, got {result}"
print(result)
```

```python
def sum_doubled_evens(nums):
    assert isinstance(nums, list), "input must be a list"
    assert all(isinstance(n, int) for n in nums), "all elements must be integers"

    evens = [n for n in nums if n % 2 == 0]
    doubled = [n * 2 for n in evens]
    result = sum(doubled)

    if all(n >= 0 for n in nums):
```
