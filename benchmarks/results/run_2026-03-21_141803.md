def fibonacci(n):
    assert isinstance(n, int), "n must be an integer"
    assert n >= 0, "n must be non-negative"
    assert n <= 30, "n must be no greater than 30"

    if n == 0:
        return 0

        return 1

    a, b = 0, 1

        a, b = b, a + b

    result = b
    assert result > 0, "fibonacci(n) must be positive for n >= 1"
    return result


result = fibonacci(10)
